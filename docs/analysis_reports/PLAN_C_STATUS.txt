BASE DE DATOS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Ubicación: bitacora/memory.db (SQLite)

Tablas:
  experiences
    ├─ id (PK)
    ├─ objective (TEXT)
    ├─ evaluation_status (TEXT: success/partial/failed)
    ├─ evaluation_score (REAL: 0.0-1.0)
    ├─ parameters (TEXT: JSON)
    ├─ metrics_passed/total (INT)
    ├─ issues (TEXT: JSON list)
    ├─ recommendations (TEXT: JSON list)
    ├─ timestamp (TEXT: ISO 8601)
    └─ similarity_hash (TEXT)

  learnings
    ├─ id (PK)
    ├─ pattern (TEXT UNIQUE)
    ├─ rule (TEXT)
    ├─ confidence (REAL)
    ├─ frequency (INT)
    ├─ applicable_to (TEXT: JSON list)
    └─ created_at (TEXT)

Índices:
  ✅ idx_status (experiences.evaluation_status)
  ✅ idx_hash (experiences.similarity_hash)
